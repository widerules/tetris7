class Dasher(text : java.lang.String)
extends AnyRef
Class used via an implicit conversion to enable describers and examples in a Spec subclass to be specified by string followed by dashes, such as: 
   "A Stack (when empty)" -- {
     "should not be full" - {
       assert(!stack.full)
     }
   }
